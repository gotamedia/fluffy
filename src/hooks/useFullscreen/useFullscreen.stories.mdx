## About
useFullscreen is a helper hook that handles fullscreen on any HTML element as long as the browser supports the fullscreen API

## Usage
<Source
    language={'tsx'}
    code={`
import { useFullscreen } from '@gotamedia/fluffy'
\nconst Component = () => {
    const [element, setElement] = useState(null)\n
    const {
        close,
        open,
        isFullscreen,
        isSupported
    } = useFullscreen(element)\n
    return (
        <div ref={setElement}>
            <p>{'Is supported: '} {String(isSupported)}</p>
            <p>{'Is fullscreen: '} {String(isFullscreen)}</p>\n
            <button onClick={open}>{'open fullscreen'}</button>
            <button onClick={close}>{'close fullscreen'}</button>
        </div>
    )
}
`}
/>
